[docs]    def set_file(self, fname):
        if fname == None:
            self.skip = True
            return
        else:
            if not fname.endswith(".log"):
                self.skip = False
                self.fname = fname + ".log"
            elif fname.endswith(".log"):
                self.fname = fname
            with open(self.fname, "w") as f:
                f.write("Logfile Created at {} \n".format(datetime.now()))
